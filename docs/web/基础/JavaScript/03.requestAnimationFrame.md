# requestAnimationFrame

```js
    let intervalTimer = null

    const customInterval = (callback, time = 1500) => {
        let now = Date.now
        let stime = now()
        let etime = stime
        let loop = () => {
            intervalTimer = requestAnimationFrame(loop)
            etime = now()
            if (etime - stime >= time) {
                stime = now()
                etime = stime
                callback()
            }
        }
        // cancelAnimationFrame
        intervalTimer = requestAnimationFrame(loop)
    }

    window.onload = function (ev) {
        customInterval(() => { console.log(123) })
    }
```
